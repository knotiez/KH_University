----- 테이블 드랍



------------- 교수 교직원 테이블
CREATE 
 TABLE MEMBER
     (
       MEMBER_NO NUMBER CONSTRAINT PK_MEMBER PRIMARY KEY
     , MEMBER_ID VARCHAR2(30) UNIQUE NOT NULL
     , MEMBER_PWD VARCHAR2(100) NOT NULL
     , MEMBER_NAME VARCAHR2(15) NOT NULL
     , MEMBER_SSN VARCHAR2(13) NOT NULL
     , M_PHONE VARCHAR2(14) NOT NULL
     , M_EMAIL VARCHAR2(50) NOT NULL
     , M_ADDRESS VARCHAR2(200) NOT NULL
     , M_TYPE VARCHAR2(1) CHECK(M_TYPE IN('P', 'A')) NOT NULL
     , M_STATUS VARCHAR2(1) CHECK(M_STATUS IN ('Y', 'N') DEFAULT 'Y'
     );

COMMENT ON COLUMN MEMBER_NO IS '회원번호';
COMMENT ON COLUMN MEMBER_ID IS '아이디';
COMMENT ON COLUMN MEMBER_PWD IS '비밀번호';
COMMENT ON COLUMN MEMBER_NAME IS '이름';
COMMENT ON COLUMN MEMBER_SSN IS '주민등록번호';
COMMENT ON COLUMN M_PHONE IS '전화번호';
COMMENT ON COLUMN M_EMAIL IS '이메일';
COMMENT ON COLUMN M_ADDRESS IS '주소';
COMMENT ON COLUMN M_TYPE IS '교수:P 행정직원:A';
COMMENT ON COLUMN M_STATUS IS '재직 : Y 퇴직 : N';

CREATE SEQUENCE SEQ_MNO NOCACHE;
--------------------- 학생 정보
CREATE 
 TABLE STUDENT
     (
       STUDENT_NO NUMBER CONSTRAINT PK_MEMBER PRIMARY KEY
     , STUDENT_ID VARCHAR2(30) UNIQUE NOT NULL
     , STUDENT_PWD VARCHAR2(100) NOT NULL
     , STUDENT_NAME VARCAHR2(15) NOT NULL
     , STUDENT_SSN VARCHAR2(13) NOT NULL
     , S_PHONE VARCHAR2(14) NOT NULL
     , S_EMAIL VARCHAR2(50) NOT NULL
     , S_ADDRESS VARCHAR2(200) NOT NULL
     , S_STATUS VARCHAR2(1) CHECK(S_STATUS IN ('Y', 'N','H','Z','J') DEFAULT 'Y'
     , ENROLL_DATE DATE DEFAULT SYSDATE NOT NULL
     , ABSENCE_DATE DATE 
     , GRAD_DATE DATE
     );

COMMENT ON COLUMN STUDENT_NO IS '회원번호';
COMMENT ON COLUMN STUDENT_ID IS '아이디';
COMMENT ON COLUMN STUDENT_PWD IS '비밀번호';
COMMENT ON COLUMN STUDENT_NAME IS '이름';
COMMENT ON COLUMN STUDENT_SSN IS '주민등록번호';
COMMENT ON COLUMN S_PHONE IS '전화번호';
COMMENT ON COLUMN S_EMAIL IS '이메일';
COMMENT ON COLUMN S_ADDRESS IS '주소';
COMMENT ON COLUMN S_STATUS IS '학생 상태: 재학 Y 자퇴 N 휴학 H ㅈㅔ적 Z 졸업 J ';
COMMENT ON COLUMN ENROLL_DATE IS '입학년도'
COMMENT ON COLUMN ABSENCE_DATE IS '휴학년도'
COMMENT ON COLUMN GRAD_DATE IS '졸업년도'


CREATE SEQUENCE SEQ_SNO NOCACHE;
-------------------- 과목
CREATE
 TABLE CLASS
     (
       CLASS_NO NUMBER CONSTRAINT PK_CLASS PRIMARY KEY
     , PROF_NO NUMBER CONSTRAINT FK_CLASS REFERENCES MEMBER
     , CLASS_NAME VARCHAR2(50) NOT NULL
     , CLASS_TYPE VARCHAR2(2) CHECK(CLASS_TYPE IN ('C1', 'C2', 'C3', 'C4'))
     , CLASS_SCORE NUMBER NOT NULL
     );
     
COMMENT ON COLUMN CLASS_NO IS '강의 번호';
COMMENT ON COLUMN PROF_NO IS '강의 담당 교수 번호';
COMMENT ON COLUMN CLASS_NAME IS '강의 제목';
COMMENT ON COLUMN CLASS_TYPE IS 'C1:필수전공 C2:선택전공 C3:필수교양 C4: 선택교양';
COMMENT ON COLUMN CLASS_SCORE IS '이수학점';

CREATE SEQUENCE SEQ_CLASS NOCACHE;

--------- 과목상세
CREATE 
 TABLE CLASS_DETAIL
     (
       REF_CNO NUMBER CONSTRAINT PK_CLASS_DETAIL PRIMARY KEY REFERENCES CLASS
     , CLASS_DAY VARCHAR2(9) NOT NULL
     , CLASS_TIME VARCHAR2(20) NOT NULL
     );
     
COMMENT ON COLUMN REF_CNO IS '강의 번호';
COMMENT ON COLUMN CLASS_DAY IS '강의 요일';
COMMENT ON COLUMN CLASS_TIME IS '강의 시간';

---------- 성적
CREATE
 TABLE GRADE
     (
       CLASS_NO NUMBER CONSTRAINT FK_GRADE_01 REFERENCES CLASS
     , STUDENT_NO NUMBER CONSTRAINT FK_GRADE_02 REFERENCES STUDENT
     , GRADE VARCHAR2(3) NOT NULL
     , TERM_NO VARCHAR2(6) NOT NULL
     , CONSTRAINT PK_GRADE PRIMARY KEY(CLASS_NO, STUDENT_NO)
     );
COMMENT ON COLUMN CLASS_NO IS '강의 번호';
COMMENT ON COLUMN STUDENT_NO IS '학생 번호';
COMMENT ON COLUMN GRADE IS '학점';
COMMENT ON COLUMN TERM_NO IS '학기 번호';

------------- 커뮤니티

CREATE
 TABLE BOARD
     (
       BOARD_NO NUMBER CONSTRAINT PK_BOARD PRIMARY KEY
     , BOARD_WRITER NUMBER CONSTRAINT FK_STUDENT REFERENCES STUDENT
     , BOARD_TITLE VARCHAR2(100) NOT NULL
     , BOARD_CONTENT CARCHAR2(4000) NOT NULL
     , COUNT NUMBER DEFAULT 0
     , CREATE_DATE DATE DEFAULT SYSDATE
     , B_STATUS VARCHAR2(1) DEFAULT 'Y' CHECK(B_STATUS IN ('Y','N'))
     , ORIGIN_NAME VARCHAR2(100)
     , CHANGE_NAME VARCHAR2(100)
     );
     
     
COMMENT ON COLUMN BOARD_NO IS '글 번호';
COMMENT ON COLUMN BOARD_WRITER IS '작성자 학생 번호';
COMMENT ON COLUMN BOARD_TITLE IS '글 제목';
COMMENT ON COLUMN BOARD_CONTENT IS '글 내용';
COMMENT ON COLUMN COUNT IS '조회수';
COMMENT ON COLUMN CREATE_DATE IS '작성일';
COMMENT ON COLUMN B_STATUS IS 'Y 전시 N 삭제';
COMMENT ON COLUMN ORIGIN_NAME IS '첨부파일 원래이름';
COMMENT ON COLUMN CHANGE_NAME IS '첨부파일 바뀐이름';

CREATE SEQUENCE SEQ_BNO NOCACHE;

--------------- 댓글
CREATE
 TABLE REPLY
     (
       REPLY_NO NUMBER CONSTRAINT PK_REPLY PRIMARY KEY
     , REF_BNO NUMBER CONSTRAINT FK_REPLY_01 REFERENCES BOARD
     , REPLY_WRITER NUMBER CONSTRAINT FK_REPLY_02 REFERENCES STUDENT
     , REPLY_CONTENT VARCHAR2(400) NOT NULL
     , CREATE_DATE DATE DEFAULT SYSDATE
     , R_STATUS VARCHAR2(1) CHECK(R_STATUS IN ('Y','N')
     );
     
COMMENT ON COLUMN REPLY_NO IS '댓글번호';
COMMENT ON COLUMN REF_BNO IS '게시글번호';
COMMENT ON COLUMN REPLY_WRITER IS '작성자 학생번호';
COMMENT ON COLUMN REPLY_CONTENT IS '댓글내용';
COMMENT ON COLUMN CREATE_DATE IS '작성날짜';
COMMENT ON COLUMN R_STATUS IS '댓글 상태';

CREATE SEQUENCE SEQ_RNO NOCACHE;

-------------- 등록금 고지서
CREATE
 TABLE TUITION
     (
       TT_NO NUMBER CONSTRAINT PK_TUITION PRIMARY KEY
     , STUDENT_NO NUMBER CONSTRAINT FK_TUITION REFERENCES STUDENT
     , STU_GRADE NUMBER NOT NULL
     , TERM_NO VARCHAR2(6) NOT NULL
     , TUITION NUMBER NOT NULL
     , SCHOLARSHIP NUMBER NOT NULL
     );
     
COMMENT ON COLUMN TT_NO IS '고지서 번호';
COMMENT ON COLUMN STUDENT_NO IS '학번';
COMMENT ON COLUMN STU_GRADE IS '학년';
COMMENT ON COLUMN TERM_NO IS '학기';
COMMENT ON COLUMN TUITION IS '등록금';
COMMENT ON COLUMN SCHOLARSHIP IS '수혜금';

CREATE SEQUENCE SEQ_TNO NOCACHE;

---------------- 등록금 납부
CREATE 
 TABLE TUITION_PAYMENT
     (
       TP_NO NUMBER CONSTRAINT PK_TUITION_PAYMENT PRIMARY KEY
     , TT_NO NUMBER CONSTRAINT FK_TUITION_PAYMENT REFERENCES TUITION
     , TP_DATE DATE NOT NULL
     , TP_TOTAL_PRICE NUMBER
     );

COMMENT ON COLUMN TP_NO IS '등록금 납부 번호';

COMMENT ON COLUMN TP_DATE IS '납부일 (기한)';

CREATE SEQUENCE SEQ_TPNO NOCACHE;
